2) Add this to your global CSS (e.g., index.css)
/* One-page A4 print layout shared by lab request/report */
.rx-print {
  width: 210mm;
  min-height: calc(297mm - 24mm);  /* 24mm = 12mm top + 12mm bottom margins */
  padding: 10mm;
  box-sizing: border-box;
  display: flex;
  flex-direction: column;
}

/* Keep footer pinned to bottom */
.rx-print .mt-auto { margin-top: auto !important; }

@media print {
  @page { size: A4; margin: 12mm; }
  html, body { margin: 0; padding: 0; }
  body * { visibility: hidden; }

  /* Show only the active print section (whichever is mounted) */
  #lab-request-print, #lab-request-print * { visibility: visible; }
  #lab-report-print, #lab-report-print * { visibility: visible; }

  /* Place the page at the top-left and prevent overflow-induced extra pages */
  #lab-request-print,
  #lab-report-print {
    position: absolute;
    inset: 0;
    overflow: hidden;
    page-break-after: avoid;
    page-break-before: avoid;
  }

  /* Hide UI chrome/buttons */
  .no-print { display: none !important; }
}


Why this works: we avoid 100vh (which often prints taller than the page once margins/padding are added) and size the content in millimetres inside the @page margins. That prevents the tiny overflow that causes a trailing blank page.

3) Make the print toggles unmount after printing finishes

Replace both printLabRequest and printLabReport with:

const printLabRequest = () => {
  if (!selectedPatient || selectedTests.length === 0) {
    toast({ title: "Error", description: "Please select a patient and tests before printing", variant: "destructive" });
    return;
  }
  setShowLabRequest(true);
  setTimeout(() => {
    const done = () => setShowLabRequest(false);
    window.addEventListener("afterprint", done, { once: true });
    window.print();
  }, 50);
};

const printLabReport = () => {
  if (!selectedLabTest) {
    toast({ title: "Error", description: "Please select a lab test to print the report", variant: "destructive" });
    return;
  }
  setShowLabReport(true);
  setTimeout(() => {
    const done = () => setShowLabReport(false);
    window.addEventListener("afterprint", done, { once: true });
    window.print();
  }, 50);
};


That ensures the print DOM stays mounted until the print job completes (no mid-process unmounts that can yield blank pages).
